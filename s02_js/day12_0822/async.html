                <script>
                    const root = document.querySelector('#root');
        
                    const list = [
                        'A. 라면 장보기', 
                        'B. 물 끓이기', 
                        'C. 끓는 물에 라면, 스프 넣고, 익히기',
                        'D. 완성'
                    ]

                    root.innerHTML = '==== 동기방식 ===='+ '<br>';
                    for (let i = 0; i < list.length; i++) {
                        root.innerHTML += list[i] + '<br>';
                    }

                    root.innerHTML += '==== 비동기 방식 ====' + '<br>';
                    setTimeout(() => {
                        root.innerHTML += list[0] + '<br>';
                    })

                    for (let i = 1; i < list.length; i++) {
                        root.innerHTML += list[i] + '<br>';
                    }       
                </script>
