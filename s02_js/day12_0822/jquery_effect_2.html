Javascript

"500px" 이동

"50px"씩 이동 

                <script>
                    $(function() {
                        $('.btn1').on('click', function() {
                            $('.txt1').animate({
                                marginLeft: '500px',
                                fontSize: '30px'
                            },
                            2000, 'linear', function() {
                                alert('모션 완료!');
                            });                
                        });

                        $('.btn2').on('click', () => {
                            $('.txt2').animate({
                                marginLeft: '+=50px'
                            }, 1000);
                        });
                    });
                </script>
