                <script>
                    const root = document.querySelector('#root');

                    for(let i = 1; i <= 100; i++) {
                        if(i % 5 == 0 && i% 7 != 0) {
                            root.innerHTML += `<span class = 'red'> ${i} </span>`;
                        } else if (i % 7 == 0 && i % 5 != 0) {
                            root.innerHTML += `<span class = 'green'> ${i} </span>`;
                        } else if (i % 5 == 0 && i % 7 == 0) {
                            root.innerHTML += `<span class = 'aqua'> ${i} </span>`;
                        } else {
                            root.innerHTML += `<span> ${i} </span>`;
                        }
                    }
                </script>
