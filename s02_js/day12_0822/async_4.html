                <script>
                    const root = document.querySelector('#root');
        
                    function rocketReady(rocketName) {
                        return new Promise((resolve, reject) => {
                            setTimeout(() => {
                                const result = Math.random() > 0.3;
                                root.innerHTML += `result: ${result} <br>`;

                                if(result) {
                                    resolve(`${rocketName} 발사 완료`);
                                } else {
                                    reject(`${rocketName} 발사 실패`);
                                }
                            }, 3000)
                        });
                    }

                    rocketReady('나로호')
                        .then((response) => {
                            root.innerHTML += `성공: ${response} <br>`;
                        })
                        .catch((response) => {
                            root.innerHTML += `실패: ${response} <br>`;
                        })
                        .finally((response) => {
                            root.innerHTML += '우주 발사체 테스트 완료';
                        });
                </script>
